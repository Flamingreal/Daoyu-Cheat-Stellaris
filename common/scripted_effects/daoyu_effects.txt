		event_target:daoyu_target_planet = {
			solar_system = {
				if = {
					limit = { has_global_flag = daoyu_yes_hyperlane }
					spawn_system = { min_distance = $DISTANCE_VALUE$ max_distance = $DISTANCE_VALUE$ min_orientation_angle = $ANGLE_VALUE$ max_orientation_angle = $ANGLE_VALUE$ max_jumps = 0 hyperlane = yes initializer = $WHAT$ effect = { save_global_event_target_as = daoyu_new_initializer } }
				}
				else = { spawn_system = { min_distance = $DISTANCE_VALUE$ max_distance = $DISTANCE_VALUE$ min_orientation_angle = $ANGLE_VALUE$ max_orientation_angle = $ANGLE_VALUE$ max_jumps = 0 hyperlane = no initializer = $WHAT$ effect = { save_global_event_target_as = daoyu_new_initializer } } }
			}
		}
	}
	if = {
		limit = { has_global_flag = daoyu_spawn_megastructure }
		event_target:daoyu_target_planet.solar_system = {
			spawn_megastructure = { type = $WHAT$ orbit_distance = $DISTANCE_VALUE$ orbit_angle = $ANGLE_VALUE$ owner = root graphical_culture = root init_effect = { save_event_target_as = daoyu_megastructure } }
		}
	}
	if = {
		limit = { has_global_flag = daoyu_spawn_planet }
		event_target:daoyu_target_planet = { save_global_event_target_as = daoyu_prev_target_planet }
		event_target:daoyu_target_planet.solar_system = {
			if = {
				limit = { has_global_flag = daoyu_has_ring }
				spawn_planet = { generate_random_name = yes class = $WHAT$ location = event_target:daoyu_target_planet orbit_location = yes orbit_distance_offset = $DISTANCE_VALUE$ orbit_angle_offset = $ANGLE_VALUE$ has_ring = yes init_effect = { save_global_event_target_as = daoyu_target_planet } }
			}

		}
	}
